58   /**
59    * Executes the Flux query against the InfluxDB and asynchronously stream [[FluxRecord]]s to [[Stream]].
60    *
61    * @param query the flux query to execute
62    * @return the stream of [[FluxRecord]]s
63    */
64   override def query(query: String): Source[FluxRecord, NotUsed] = {
65 
66     Arguments.checkNonEmpty(query, "query")
67 
68     Source
69       .single(query)
70       .map(it => fluxService.query(createBody(AbstractQueryApi.DEFAULT_DIALECT.toString, it)))
71       .flatMapConcat(queryCall => {
72         Source.queue[FluxRecord](bufferSize, overflowStrategy)
73           .mapMaterializedValue(queue => {
74 
75             val eventualDone = queue.watchCompletion()
76             val consumer = new FluxCsvParser.FluxResponseConsumer() {
77 
78               override
79               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull table: FluxTable): Unit = {
80               }
81 
82               override
83               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull record: FluxRecord): Unit = {
84                 if (eventualDone.isCompleted) {
85                   cancellable.cancel()
86                 } else {
87                   queue.offer(record)

89               }
90             }
91 
92             val onError = asJavaConsumer[Throwable](t =>  queue.fail(t))

94             this.query(queryCall, consumer, onError, () => queue.complete, true)
