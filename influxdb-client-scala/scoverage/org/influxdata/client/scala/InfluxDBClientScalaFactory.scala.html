24 import akka.stream.OverflowStrategy

30 /**
31  * The Factory that creates a instance of a Flux client.
32  * 
33  * @author Jakub Bednar (bednar@github) (05/11/2018 10:10)
34  */
35 object InfluxDBClientScalaFactory {
36 
37   /**
38    * Create a instance of the InfluxDB 2.0 reactive client.
39    *
40    * @param url the url to connect to the InfluxDB
41    * @return client
42    * @see [[InfluxDBClientOptions.Builder]]
43    */
44   @Nonnull def create(url: String): InfluxDBClientScala = {
45 
46     val options = InfluxDBClientOptions.builder()
47       .url(url)
48       .build()
49 
50     create(options)
51   }
52 
53   /**
54    * Create a instance of the InfluxDB 2.0 reactive client.
55    *
56    * @param url      the url to connect to the InfluxDB
57    * @param username the username to use in the basic auth
58    * @param password the password to use in the basic auth
59    * @return client
60    * @see [[InfluxDBClientOptions.Builder]]
61    */
62   @Nonnull def create(url: String, username: String, password: Array[Char]): InfluxDBClientScala = {
63 
64     val options = InfluxDBClientOptions.builder()
65       .url(url)
66       .authenticate(username, password)
67       .build()
68 
69     create(options)
70   }
71 
72   /**
73    * Create a instance of the InfluxDB 2.0 reactive client.
74    *
75    * @param url   the url to connect to the InfluxDB
76    * @param token the token to use for the authorization
77    * @return client
78    * @see [[InfluxDBClientOptions.Builder]]
79    */
80   @Nonnull def create(url: String, token: Array[Char]): InfluxDBClientScala = {
81 
82     val options = InfluxDBClientOptions.builder()
83       .url(url)
84       .authenticateToken(token)
85       .build()
86 
87     create(options)
88   }
89 
90   /**
91    * Create a instance of the InfluxDB 2.0 reactive client.
92    *
93    * @param options          the connection configuration
94    * @param bufferSize       size of a buffer for incoming responses. Default 10000.
95    * @param overflowStrategy Strategy that is used when incoming response cannot fit inside the buffer.
96    *                         Default [[akka.stream.OverflowStrategies.Backpressure]].
97    * @see [[akka.stream.scaladsl.Source#queue(int, akka.stream.OverflowStrategy)]]
98    * @return client
99    */
100   @Nonnull def create(@Nonnull options: InfluxDBClientOptions,
101                       @Nonnull bufferSize: Int = 10000,
102                       @Nonnull overflowStrategy: OverflowStrategy = OverflowStrategy.backpressure): InfluxDBClientScala = {
103 
104     Arguments.checkNotNull(options, "InfluxDBClientOptions")
105 
106     new InfluxDBClientScalaImpl(options, bufferSize, overflowStrategy)
107   }
108 }
