35 class InfluxDBClientScalaImpl(@Nonnull options: InfluxDBClientOptions,

39    * Get the Query client.
40    *
41    * @return the new client instance for the Query API

44 
45   /**
46    * Get the health of an instance.
47    *
48    * @return health of an instance
49    */

51 
52   /**
53    * Gets the [[LogLevel]] that is used for logging requests and responses.
54    *
55    * @return the [[LogLevel]] that is used for logging requests and responses
56    */
57   override def getLogLevel: LogLevel = getLogLevel(loggingInterceptor)
58 
59   /**
60    * Sets the log level for the request and response information.
61    *
62    * @param logLevel the log level to set.
63    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
64    */
65   override def setLogLevel(logLevel: LogLevel): InfluxDBClientScala = {
66     setLogLevel(loggingInterceptor, logLevel)
67     this
68   }
69 
70   /**
71    * Enable Gzip compress for http request body.
72    *

74    *
75    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
76    */
77   override def enableGzip(): InfluxDBClientScala = {
78     gzipInterceptor.enableGzip()
79     this
80   }
81 
82   /**

84    *
85    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
86    */
87   override def disableGzip(): InfluxDBClientScala = {
88     gzipInterceptor.disableGzip()
89     this
90   }
91 
92   /**
93    * Returns whether Gzip compress for http request body is enabled.
94    *
95    * @return true if gzip is enabled.
96    */
97   override def isGzipEnabled: Boolean = gzipInterceptor.isEnabledGzip
