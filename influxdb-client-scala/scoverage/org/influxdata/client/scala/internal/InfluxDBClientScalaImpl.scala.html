36 class InfluxDBClientScalaImpl(@Nonnull options: InfluxDBClientOptions,
37                               @Nonnull val bufferSize: Int,
38                               @Nonnull val overflowStrategy: OverflowStrategy) extends AbstractInfluxDBClient(options) with InfluxDBClientScala {
39   /**
40    * Get the Query client.
41    *
42    * @return the new client instance for the Query API
43    */
44   override def getQueryScalaApi(): QueryScalaApi = new QueryScalaApiImpl(retrofit.create(classOf[QueryService]), bufferSize, overflowStrategy)
45 
46   /**
47    * Get the health of an instance.
48    *
49    * @return health of an instance
50    */
51   override def health: Check = health(healthService.getHealth(null))
52 
53   /**
54    * Gets the [[LogLevel]] that is used for logging requests and responses.
55    *
56    * @return the [[LogLevel]] that is used for logging requests and responses
57    */
58   override def getLogLevel: LogLevel = getLogLevel(loggingInterceptor)
59 
60   /**
61    * Sets the log level for the request and response information.
62    *
63    * @param logLevel the log level to set.
64    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
65    */
66   override def setLogLevel(logLevel: LogLevel): InfluxDBClientScala = {
67     setLogLevel(loggingInterceptor, logLevel)
68     this
69   }
70 
71   /**
72    * Enable Gzip compress for http request body.
73    *
74    * Currently only the "Write" endpoint supports the Gzip compression.
75    *
76    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
77    */
78   override def enableGzip(): InfluxDBClientScala = {
79     gzipInterceptor.enableGzip()
80     this
81   }
82 
83   /**
84    * Disable Gzip compress for http request body.
85    *
86    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
87    */
88   override def disableGzip(): InfluxDBClientScala = {
89     gzipInterceptor.disableGzip()
90     this
91   }
92 
93   /**
94    * Returns whether Gzip compress for http request body is enabled.
95    *
96    * @return true if gzip is enabled.
97    */
98   override def isGzipEnabled: Boolean = gzipInterceptor.isEnabledGzip
