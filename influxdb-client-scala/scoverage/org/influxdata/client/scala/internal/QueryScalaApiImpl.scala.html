76 
77     Arguments.checkNotNull(query, "query")
78     Arguments.checkNonEmpty(orgID, "orgID")

85       .flatMapConcat(queryCall => {

87           .mapMaterializedValue(queue => {
88 
89             val eventualDone = queue.watchCompletion()

91 
92               override
93               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull table: FluxTable): Unit = {

98                 if (eventualDone.isCompleted) {
99                   cancellable.cancel()
100                 } else {
101                   queue.offer(record)

103               }
104             }
