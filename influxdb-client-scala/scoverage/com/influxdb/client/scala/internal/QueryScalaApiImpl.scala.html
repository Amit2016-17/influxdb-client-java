105    * @param org   specifies the source organization

108   override def query(query: Query, org: String): Source[FluxRecord, NotUsed] = {
109 
110     Arguments.checkNotNull(query, "query")
111     Arguments.checkNonEmpty(org, "org")
112 
113     Source
114       .single(query)
115       .map(_ => service
116         .postQueryResponseBody(null, "application/json",
117           null, org, null, query))
118       .flatMapConcat(queryCall => {
119         Source.queue[FluxRecord](bufferSize, overflowStrategy)
120           .mapMaterializedValue(queue => {
121 
122             val eventualDone = queue.watchCompletion()
123             val consumer = new FluxCsvParser.FluxResponseConsumer() {
124 
125               override
126               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull table: FluxTable): Unit = {
127               }
128 
129               override
130               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull record: FluxRecord): Unit = {
131                 if (eventualDone.isCompleted) {
132                   cancellable.cancel()
133                 } else {
134                   queue.offer(record)
135                 }
136               }
137             }
138 
139             val onError = asJavaConsumer[Throwable](t => queue.fail(t))
140 
141             this.query(queryCall, consumer, onError, new Runnable() { def run(): Unit = queue.complete() }, true)
142           })
143       })
144   }
