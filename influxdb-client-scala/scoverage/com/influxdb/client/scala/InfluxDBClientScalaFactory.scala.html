24 import akka.stream.OverflowStrategy

30 /**
31  * The Factory that creates an instance of a Flux client.
32  * 
33  * @author Jakub Bednar (bednar@github) (05/11/2018 10:10)
34  */
35 object InfluxDBClientScalaFactory {
36 
37 
38   /**
39    * Create an instance of the InfluxDB 2.0 client that is configured via `influx2.properties`.
40    * The `influx2.properties` has to be located on classpath.
41    *
42    * @return client
43    */
44   @Nonnull def create(): InfluxDBClientScala = {
45 
46     val options = InfluxDBClientOptions.builder()
47       .loadProperties()
48       .build()
49 
50     create(options)
51   }
52 
53   /**
54    * Create an instance of the InfluxDB 2.0 client. The url could be a connection string with various configurations.
55    *
56    * e.g.: "http://localhost:8086?readTimeout=5000&amp;connectTimeout=5000&amp;logLevel=BASIC
57    *
58    * @param connectionString connection string with various configurations.
59    * @return client
60    */
61   @Nonnull def create(connectionString: String): InfluxDBClientScala = {
62 
63     val options = InfluxDBClientOptions.builder()
64       .url(connectionString)
65       .build()
66 
67     create(options)
68   }
69 
70   /**
71    * Create an instance of the InfluxDB 2.0 reactive client.
72    *
73    * @param url      the url to connect to the InfluxDB
74    * @param username the username to use in the basic auth
75    * @param password the password to use in the basic auth
76    * @return client
77    * @see [[InfluxDBClientOptions.Builder]]
78    */
79   @Nonnull def create(url: String, username: String, password: Array[Char]): InfluxDBClientScala = {
80 
81     val options = InfluxDBClientOptions.builder()
82       .url(url)
83       .authenticate(username, password)
84       .build()
85 
86     create(options)
87   }
88 
89   /**
90    * Create an instance of the InfluxDB 2.0 reactive client.
91    *
92    * @param url   the url to connect to the InfluxDB
93    * @param token the token to use for the authorization
94    * @return client
95    * @see [[InfluxDBClientOptions.Builder]]
96    */
97   @Nonnull def create(url: String, token: Array[Char]): InfluxDBClientScala = {
98 
99     create(url, token, null)
100   }
101 
102   /**
103    * Create an instance of the InfluxDB 2.0 reactive client.
104    *
105    * @param url   the url to connect to the InfluxDB
106    * @param token the token to use for the authorization
107    * @param org      the name of an organization
108    * @return client
109    * @see [[InfluxDBClientOptions.Builder]]
110    */
111   @Nonnull def create(url: String, token: Array[Char], org: String): InfluxDBClientScala = {
112 
113     create(url, token, org, null)
114   }
115 
116   /**
117    * Create an instance of the InfluxDB 2.0 reactive client.
118    *
119    * @param url   the url to connect to the InfluxDB
120    * @param token the token to use for the authorization
121    * @param org      the name of an organization
122    * @param bucket   the name of a bucket
123    * @return client
124    * @see [[InfluxDBClientOptions.Builder]]
125    */
126   @Nonnull def create(url: String, token: Array[Char], org: String, bucket: String): InfluxDBClientScala = {
127 
128     val options = InfluxDBClientOptions.builder()
129       .url(url)
130       .authenticateToken(token)
131       .org(org)
132       .bucket(bucket)
133       .build()
134 
135     create(options)
136   }
137 
138   /**
139    * Create an instance of the InfluxDB 2.0 reactive client.
140    *
141    * @param options          the connection configuration
142    * @param bufferSize       size of a buffer for incoming responses. Default 10000.
143    * @param overflowStrategy Strategy that is used when incoming response cannot fit inside the buffer.
144    *                         Default [[akka.stream.OverflowStrategies.Backpressure]].
145    * @see [[akka.stream.scaladsl.Source#queue(int, akka.stream.OverflowStrategy)]]
146    * @return client
147    */
148   @Nonnull def create(@Nonnull options: InfluxDBClientOptions,
149                       @Nonnull bufferSize: Int = 10000,
150                       @Nonnull overflowStrategy: OverflowStrategy = OverflowStrategy.backpressure): InfluxDBClientScala = {
151 
152     Arguments.checkNotNull(options, "InfluxDBClientOptions")
153 
154     new InfluxDBClientScalaImpl(options, bufferSize, overflowStrategy)
155   }
156 }
