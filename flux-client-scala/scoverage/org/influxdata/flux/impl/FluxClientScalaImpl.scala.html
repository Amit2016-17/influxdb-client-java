56   /**
57    * Executes the Flux query against the InfluxDB and asynchronously stream [[FluxRecord]]s to [[Stream]].
58    *
59    * @param query the flux query to execute
60    * @return the stream of [[FluxRecord]]s
61    */
62   override def query(query: String): Source[FluxRecord, NotUsed] = {
63 
64     Arguments.checkNonEmpty(query, "query")
65 
66     Source
67       .single(query)
68       .map(it => fluxService.query(createBody(AbstractQueryClient.DEFAULT_DIALECT.toString, it)))
69       .flatMapConcat(queryCall => {
70         Source.queue[FluxRecord](bufferSize, overflowStrategy)
71           .mapMaterializedValue(queue => {
72 
73             val eventualDone = queue.watchCompletion()
74             val consumer = new FluxCsvParser.FluxResponseConsumer() {
75 
76               override
77               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull table: FluxTable): Unit = {
78               }
79 
80               override
81               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull record: FluxRecord): Unit = {
82                 if (eventualDone.isCompleted) {
83                   cancellable.cancel()
84                 } else {
85                   queue.offer(record)
86                 }
87               }
88             }
89 
90             val onError = asJavaConsumer[Throwable](t =>  queue.fail(t))
91 
92             this.query(queryCall, consumer, onError, () => queue.complete, true)
93           })
94     })
95   }

99    *

102    * @tparam M the type of the measurement (POJO)

164   /**
165    * Check the status of InfluxDB Server.
166    *
167    * @return `true` if server is healthy otherwise return `false`
168    */
169   override def ping: Boolean = {
170     try {
171       fluxService.ping().execute().isSuccessful
172     } catch {
173       case e: IOException =>
174         LOG.log(Level.WARNING, "Ping request wasn't successful", e)

176     }
