53   /**
54    * Executes the Flux query against the InfluxDB and asynchronously stream [[FluxRecord]]s to [[Stream]].
55    *
56    * @param query the flux query to execute
57    * @return the stream of [[FluxRecord]]s
58    */
59   override def query(query: String): Source[FluxRecord, NotUsed] = {
60 
61     Arguments.checkNonEmpty(query, "query")
62 
63     Source
64       .single(query)
65       .map(it => fluxService.query(createBody(AbstractQueryClient.DEFAULT_DIALECT.toString, it)))
66       .flatMapConcat(queryCall => {
67         Source.queue[FluxRecord](bufferSize, overflowStrategy)
68           .mapMaterializedValue(queue => {
69 
70             val eventualDone = queue.watchCompletion()
71             val consumer = new FluxCsvParser.FluxResponseConsumer() {
72 
73               override
74               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull table: FluxTable): Unit = {
75               }
76 
77               override
78               def accept(index: Int, @Nonnull cancellable: Cancellable, @Nonnull record: FluxRecord): Unit = {
79                 if (eventualDone.isCompleted) {
80                   cancellable.cancel()
81                 } else {
82                   queue.offer(record)
83                 }
84               }
85             }
86 
87             val onError = asJavaConsumer[Throwable](t =>  queue.fail(t))
88 
89             this.query(queryCall, consumer, onError, () => queue.complete, true)
90           })
91     })
92   }

94   /**
95    * Executes the Flux query against the InfluxDB and asynchronously stream measurements to [[Stream]].
96    *
97    * @param query           the flux query to execute
98    * @param measurementType the measurement (POJO)
99    * @tparam M the type of the measurement (POJO)
100    * @return the stream of measurements
101    */
102   override def query[M](query: String, measurementType: Class[M]): Source[M, NotUsed] = {
103 
104     Arguments.checkNonEmpty(query, "query")
105     Arguments.checkNotNull(measurementType, "measurementType")
106 
107     this.query(query).map(t => resultMapper.toPOJO(t, measurementType))
108   }

161   /**
162    * Check the status of InfluxDB Server.
163    *
164    * @return `true` if server is healthy otherwise return `false`
165    */
166   override def ping: Boolean = {
167     try {
168       fluxService.ping().execute().isSuccessful
169     } catch {
170       case e: IOException =>
171         LOG.log(Level.WARNING, "Ping request wasn't successful", e)
172         false
173     }
174   }

176   /**

178    *
179    * @return the version String, otherwise unknown.
180    */
181   override def version: String = {
182     try {
183       val response = fluxService.ping().execute()
184       getVersion(response)
185     } catch {
186       case e: IOException => throw new InfluxException(e)
187     }
188   }
