30 /**
31  * The Factory that creates a instance of a Flux client.
32  * 
33  * @author Jakub Bednar (bednar@github) (05/11/2018 10:10)
34  */
35 object FluxClientScalaFactory {
36 
37   /**
38    * Create a instance of the Flux client.
39    *
40    * @param connectionString the connectionString to connect to InfluxDB.
41    * @return client
42    * @see [[org.influxdata.flux.option.FluxConnectionOptions#builder(java.lang.String)]]
43    */
44   @Nonnull def create(@Nonnull connectionString: String): FluxClientScala = {
45     val options = FluxConnectionOptions.builder(connectionString).build
46     create(options)
47   }
48 
49   /**
50    * Create a instance of the Flux client.
51    *
52    * @param options          the connection configuration
53    * @param bufferSize       size of a buffer for incoming responses. Default 10000.
54    * @param overflowStrategy Strategy that is used when incoming response cannot fit inside the buffer.
55    *                         Default [[akka.stream.OverflowStrategies.Backpressure]].
56    * @see [[akka.stream.scaladsl.Source#queue(int, akka.stream.OverflowStrategy)]]
57    * @return client
58    */
59   @Nonnull def create(@Nonnull options: FluxConnectionOptions,
60                       @Nonnull bufferSize: Int = 10000,
61                       @Nonnull overflowStrategy: OverflowStrategy = OverflowStrategy.backpressure): FluxClientScala = {
62 
63     Arguments.checkNotNull(options, "FluxConnectionOptions")
64     Arguments.checkNotNull(overflowStrategy, "overflowStrategy")
65     Arguments.checkNotNull(bufferSize, "bufferSize")
66 
67     new FluxClientScalaImpl(options, bufferSize, overflowStrategy)
68   }
69 }
