package org.influxdata.java.client.domain;

import com.squareup.moshi.Json;

/**
 * The status of the {@link Run}.
 *
 * @author Jakub Bednar (bednar@github) (22/10/2018 09:08)
 */
public enum RunStatus {

    @Json(name = "scheduled")
    SCHEDULED,

    @Json(name = "started")
    STARTED,

    @Json(name = "failed")
    FAILED,

    @Json(name = "success")
    SUCCESS,

    @Json(name = "canceled")
    CANCELED
}
